## count vs find
```cpp
class Solution {
public:
    int numJewelsInStones(string jewels, string stones) {
        int ans = 0;
        unordered_map <char, int> um;
        for(int i = 0;i < jewels.size();++i)
            um[jewels[i]] = 0;
        
        for(int i=0;i < stones.size();++i){
            if(um.count(stones[i]))
                ans += 1;
        }
        return ans;
    }
};
```
