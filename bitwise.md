```cpp
# include <iostream>
using namespace std;
// bitwise
int main() {

	//	5 << 1 = 10 // 00101 的全部位元向左移動一位數變成 01010. 
	//	5 << 2 = 20 // 00101 的全部位元向左移動兩位數變成 10100. 
	//	5 >> 1 = 2 // 00101 的全部位元向右移動一位數變成 00010. 
	//	5 >> 2 = 1 // 00101 的全部位元向右移動一位數變成 00001.
	int tmp = 5 >> 1;
	cout << tmp;
	return 0;
}
```
